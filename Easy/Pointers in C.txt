#include <stdio.h>
void update(int *a,int *b);
void update(int *a,int *b) {
    int  p;
 p=*a;
 (*a)=(*a)+(*b);
 (*b)=(p)-(*b);
 if((*b)<0)
 {
     (*b)=-(*b);
 }
}

int main() {
    int a, b;
    int *pa = &a, *pb = &b;
    
    scanf("%d %d", &a, &b);
    update(pa, pb);
    printf("%d\n%d", a, b);

    return 0;
}
